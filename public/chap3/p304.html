An example to iterate a collection

// this example adds 1 mark to exam of everybody!
// first, select all cells in the table body
var $cells = $('table#marks tbody td:nth-child(3)');
// for each HTML element in the jQuery collection ...
for (var c of $cells) {
  // use $() to make the DOM node into a jQuery collection
  // then, we can use .text() to read content
  var mark = $(c).text();
  mark = parseInt(mark);
  $(c).text(mark+1);
}
